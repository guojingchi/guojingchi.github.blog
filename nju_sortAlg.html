1.最小交换次数

Description

 

Given an array of N distinct elementsA[ ], find the minimum number of swaps required to sort the array.Your are required to complete the function which returns an integer denoting the minimum number of swaps, required to sort the array.

 

Input

 

The first line of input contains an integer T denoting the no of test cases . Then T test cases follow . Each test case contains an integer N denoting the no of element of the array A[ ]. In the next line are N space separated values of the array A[ ] .(1<=T<=100;1<=N<=100;1<=A[] <=1000)

 

Output

 

For each test case in a new line output will be an integer denoting minimum umber of swaps that are required to sort the array.

 

Sample Input 1 

2
4
4 3 2 1
5
1 5 4 3 2

Sample Output 1

2
2

		int[] sort = new int[num];
		
		int count = 0;

		Arrays.sort(sort);

			if(sort[i]!=arr[i]) {
				int j=0;
				for(;j<num;j++) {
					if(sort[i]==arr[j])
						break;
				}
				
				int tmp = arr[i];
				arr[i] = arr[j];
				arr[j] = tmp;
				count++;
			}
			
			
		}
		
		System.out.println(count);
	}
	
	
	
	
}

4.计数排序

import java.util.*;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		
		
			int len = sc.nextInt();
			
			int[] arr = new int[len];
			
			for(int i=0;i<len;i++)
				arr[i] = sc.nextInt();
			
			
			countSort(arr);
			
		
		
	}
	
	
	public static void countSort(int[] arr) {
		
		
	   int length = arr.length;
	   
	   int[] reArr = new int[length];
		
	   int max = arr[0];
	   int min = arr[0];
	   
	   //找到最大值和最小值
	   for(int i=0;i<length;i++) {
		   if(arr[i]>max)
			   max = arr[i];
		   if(arr[i]<min)
			   min = arr[i];
	   }
	   
	   //建立从最小到最大的数组
	   int[] tmp = new int[max+min-1];
	   
	   //找到每一项的位置，数组中记录出现的次数
	   for(int i=0;i<length;i++) {
		   tmp[arr[i]-min]+=1;
	   }
		int poi = 0;
	   
	   //根据次数输出内容
	   for(int i=0;i<tmp.length;i++) {
		   
		   int count = tmp[i];
		   
		   for(int j=0;j<count;j++) {
			   reArr[poi] = min+i;
			   poi++;
		   }
			   
	   }
	   
		
		
		for(int i=0;i<reArr.length;i++) {
			if(i==length-1)
				System.out.println(reArr[i]);
			else
				System.out.print(reArr[i]+" ");
		}
		
	}

}

5.非递归快排

import java.util.*;
import java.lang.*;

public class Main {

	public static void main(String[] args) {
		// code
		Scanner sc = new Scanner(System.in);
		String[] strArr = sc.nextLine().split("\\s+");
		
		int num = Integer.valueOf(strArr[0]);
		
		int[] arr = new int[num];
		
		for(int i=0;i<num;i++) {
			arr[i] = Integer.valueOf(strArr[i+1]);
		}
		
		quickSort(arr,num);
	}
	
	
	public static void quickSort(int[] arr,int num) {
		
		//
		 int[] stack = new int[num];
		int low = 0;
		int high = num-1;
		int top=0;
		int par = partition(arr,high,low);
		
		//把小的部分的开头和结尾存起来
		if(par>low+1) {
			stack[top++] = low;
			stack[top++] = par-1;
		}
		//同理大的部分
		if(high>par+1) {
			stack[top++] = par+1;
			stack[top++]  =high;
		}
		
		//出栈处理
		while(top>0) {
			high = stack[--top];
			low = stack[--top];
			par = partition(arr,high,low);
			//把小的部分的开头和结尾存起来
			if(par>low+1) {
				stack[top++] = low;
				stack[top++] = par-1;
			}
			//同理大的部分
			if(high>par+1) {
				stack[top++] = par+1;
				stack[top++]  =high;
			}
		
		}
		
		
		for(int i=0;i<num;i++) {
			if(i==num-1)
				System.out.print(arr[i]+"\n");
			else
				System.out.print(arr[i]+" ");
		}
		
	}
	
	public static int partition(int[] arr,int high,int low) {
		int tmp = arr[low];
		
		while(low<high) {
			
			//从右向左找小的项，交换
			while(low<high&&arr[high]>=tmp)
				--high;
			if(low>=high)
				break;
			else {
				arr[low] = arr[high];
			}
			//从左向右找大的，交换
			while(low<high&&arr[low]<=tmp)
				++low;
			
			if(low>=high)
				break;
			else {
				arr[high] = arr[low];
			}
			
		}
		//最后交换分割点数值，返回分割点
		arr[low] = tmp;
		return low;
	}

}

6.非递归归并排序

import java.util.*;
import java.lang.*;

public class Main {

	public static void main(String[] args) {
		// code
		Scanner sc = new Scanner(System.in);
		String[] strArr = sc.nextLine().split("\\s+");
		
		int num = Integer.valueOf(strArr[0]);
		
		int[] arr = new int[num];
		
		for(int i=0;i<num;i++) {
			arr[i] = Integer.valueOf(strArr[i+1]);
		}
		
		mergeSort(arr,num);
	}
	
	public static void mergeSort(int[] arr,int num) {
		
		//跨度是1开始
		int span = 1;
		int[] tmp = new int[num];
		
		while(span<num) {//跨度小于数组长度
			merge(arr,tmp,span,num);
			span=span*2;
		}
		
		for(int i=0;i<num;i++) {
			if(i==num-1)
				System.out.print(arr[i]+"\n");
			else
				System.out.print(arr[i]+" ");
		}
		
	}
	
	public static void merge(int[] arr,int[] tmp,int span,int num) {
		
		int index = 0;
		
		int j;
		
		while(index+span*2-1<num) {
			//两路归并
			getCom(arr,tmp,index,index+span-1,index+span*2-1);
			//找到下一个开始的地方
			index = index+span*2;
		}
		
		if (index<num+1-span-1) {//如果剩下的序列还大于一个间隔我们也归并，最后减1是因为从0开始索引
            getCom(arr,tmp,index,index+span-1,num-1); //两两归并
        }
        else //如果只剩单个序列，我们就把它放最后，就算一直他是单个，在最后一次归并时只会剩余它和另一个，
        // 然后进行归并通过上面的if语句
        for (j=index;j<num;j++){
            tmp[j]=arr[j];
        }
		
	}
	
	public static void getCom(int[] arr,int[] tmp,int start,int mid,int end) {
		//第一部分的开头
		int sStart = start;
		//第二部分的开头
		int eStart = mid+1;
		//tmp中的位置
		int count = 0;
		
		while(start<=mid&&eStart<=end) {//开始比较
			if(arr[start]<arr[eStart]) {
				tmp[count++] = arr[start++];
			}else {
				tmp[count++] = arr[eStart++];
			}
				
		}
		
		while(start<=mid)
			tmp[count++] = arr[start++];
		while(eStart<=end)
			tmp[count++] = arr[eStart++];
		
		for(int i=0;i<count;i++)
			arr[sStart++] = tmp[i];
	
	}
	
}

7.倒置个数

import java.util.*;
import java.lang.*;

public class Main {

	public static int count=0;
	public static void main(String[] args) {
		// code
		Scanner sc = new Scanner(System.in);
		int total = Integer.valueOf(sc.nextLine());
		while(total>0) {
			count=0;
			int num = Integer.valueOf(sc.nextLine());
			int[] arr = new int[num];
			
			String[] str = sc.nextLine().split("\\s+");
			
			for(int i=0;i<num;i++)
				arr[i] = Integer.valueOf(str[i]);
			
			mergeSort(arr,0,num-1);
			System.out.println(count);
			total--;
		}
		
	}
	
	public static void mergeSort(int[] arr,int start,int end) {
		
		if(start<end) {
			int mid = (start+end)/2;
			mergeSort(arr,start,mid);
			mergeSort(arr,mid+1,end);
			merge(arr,start,mid,end);
		}
		

	
	}
	
	
	public static void merge(int[] arr,int start,int mid,int end) {
		
		int[] tmp = new int[arr.length];
		
		int sStart = start;
		
		int p1 = start;
		int p2 = mid+1;
		
		while(p1<=mid&&p2<=end) {
			
			if(arr[p1]<=arr[p2]) {
				tmp[sStart++] = arr[p1++];
			}else {
				count = count+p2-sStart;
				tmp[sStart++] = arr[p2++];
			}
		}
		
		while(p1<=mid)
			tmp[sStart++] = arr[p1++];
		while(p2<=end)
			tmp[sStart++] = arr[p2++];
		
		for(int i=start;i<=end;i++)
			arr[i]  =tmp[i];
		
		
	}

	
}

            int[] numbers = new int[arr.length];

            for(int i=0;i<numbers.length;i++){
                numbers[i] = Integer.valueOf(arr[i]);
            }

            String[] bindArr = bind.split("\\s+");

            int[] bindValue = new int[bindArr.length];

            for(int i=0;i<bindValue.length;i++){
                bindValue[i] = Integer.valueOf(bindArr[i]);
            }

            shellSort(numbers,bindValue);

            total--;
        }
        
  
	}
	 public static void shellSort(int[] numbers,int[] bind){

		 for(int i=0;i<bind.length;i++) {
			 int gap = bind[i];
			 shell(numbers,gap);
		 }
		 
		 for(int i=0;i<numbers.length;i++){
	            if(i==numbers.length-1)
	                System.out.print(numbers[i]+"\n");
	            else
	                System.out.print(numbers[i]+" ");
	        }


	 }
	 
	 public static void shell(int[] arr,int gap) {
		 
		 int tmp=0;
		 int j;
		 
		 for(int i=gap;i<arr.length;i++) {
			 tmp = arr[i];
			 
			 for(j=i-gap;j>=0;j=j-gap) {
				 if(tmp<arr[j]) {
					 arr[j+gap] = arr[j];
				 }else
					 break;
			 }
			 arr[j+gap] = tmp;
		 }
	 }

}

		int length = sc.nextInt();
		
		int[] input = new int[length];
		
		int[] times = new int[length];
		
		int[][] resultArr = new int[length][2];
		
		for(int i=0;i<length;i++) {
			input[i] = sc.nextInt();
		}
		
		for(int j=0;j<length;j++) {
			int k = 0;
			for(int i=0;i<length;i++) {
				if(input[i]==input[j])
					k++;
			}
			times[j] = k;
		}
		
		
		//排序
		for(int i=0;i<length;i++) {
			for(int j=i;j<length;j++) {
				if(times[i]<times[j]) {
					int tTemp = times[i];
					times[i] = times[j];

					input[i] = input[j];
					input[j] = aTemp;
				}else if(times[i] == times[j]) {
					if(input[i]>input[j]) {
						int tTemp = times[i];
						times[i] = times[j];
						times[j] = tTemp;
						
						int aTemp = input[i];
						input[i] = input[j];
						input[j] = aTemp;
					}
				}
			}
		}
		
		for(int i=0;i<length;i++) {
          if(i==length-1)
			System.out.println(input[i]);

            System.out.print(input[i]+" ");
		}

		}
	}
}
